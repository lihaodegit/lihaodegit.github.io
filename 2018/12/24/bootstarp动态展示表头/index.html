BootStarp动态生成表头，生成数据

一般的bootStarp-table表格定义表头都是写死的，项目突然要用到动态生成表头，研究了一下，鼓捣出来了，小小的展示一下

js

$(document).ready(function () {初始化加载
        initTable();
    });
function initTable(){
         $('#QueryTable').bootstrapTable('destroy');
         $("#QueryTable").bootstrapTable({
             ajax : function (){
                 $.ajax({
                    type: "GET",
                    url: 'test/show?tm=' + new Date().getTime(),
                    async: true,
                    data:{
                        id:id,
                        Name:Name
                    },

           contentType: "application/json;charset=utf-8",
           dataType :'JSON',
           success: function (data) {
               var columns = [];
              for (var i=0;i<data.columnnames.length;i++) {
                   columns.push({
                      field: data.listcolumns[i],//属性值
                      align: 'center',
                      title: data.columnnames[i]//表头字段
                  });
               }
               var datajson=data.list//数据
               $('#QueryTable').bootstrapTable('destroy').bootstrapTable({
                   toolbar: '#toolbar',
                   cache: false,
                   striped: false,
                   sidePagination: "client",
                   sortOrder: "desc",
                   pageSize: 10,
                   pageNumber: 1,
                   pageList: "[10, 20, 50, 100]",
                   showToggle: true,
                   showColumns: true,
                   exportDataType: "basic",
                   pagination: true,
                   showRefresh: false,
                   data: datajson,//数据赋值给data
                   columns: columns//设置好的表头
               });
            },
            error:function () {
               alert("无数据");
            }

        })
    },
         onLoadSuccess: function(){  //加载成功时执行

       },
       onClickRow:function(row, element) {
        },
   });
}
